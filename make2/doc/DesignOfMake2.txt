			Context Free Grammar of the New Makefile

inclusion :
	'include' string_list

integral_assignment :
	'extern' integer_assignment
|	string '=' integral_expression
|	string '+=' integral_expression
|	string '-=' integral_expression

integral_expression :
	integral_item
|	integral_expression '+' integral item
|	integral_expression '-' integral item

integral_item :
	integral_atom
|	integral_item '*' integral_atom
|	integral_item '/' integral_atom

integral_atom :
	INT | '${' string '}'

string_assignment :
	'extern' string_assignment  // define/modify global variable
|	string '=' string_list
|	string '+=' string_list
|	string '-=' string_list

string_list :
	string string_list
|	string
|	(empty)

string :
	string_atom
|	string_atom string

string_atom :
	STRING
|	'$(' string ')'
|	'$(' CMD string ')'
